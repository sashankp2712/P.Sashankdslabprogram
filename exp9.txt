#include <stdio.h>

int main() {
    int arr[] = {2, 3, 12, 34, 54, 67, 89};
    int n = sizeof(arr) / sizeof(arr[0]);
    int key, low = 0, high = n - 1, mid, found = 0;
    printf("Enter the number to search:\n");
    scanf("%d", &key);
    while (low <= high) {
        mid = (low + high) / 2;
        if (arr[mid] == key) {
            printf("Element found at index %d.\n", mid);
            found = 1;
            break;
        } else if (arr[mid] < key) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    if (!found) printf("Element not found.\n");
    return 0;
}
