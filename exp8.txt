#include <stdio.h>

int main() {
    int arr[] = {12, 34, 54, 2, 3, 67, 89};
    int n = sizeof(arr) / sizeof(arr[0]);
    int key, i, found = 0;
    printf("Enter the number to search:\n");
    scanf("%d", &key);
    for (i = 0; i < n; i++) {
        if (arr[i] == key) {
            printf("Element found at index %d.\n", i);
            found = 1;
            break;
        }
    }
    if (!found) printf("Element not found.\n");
    return 0;
}
