#include <stdio.h>

int main() {
    int n, num, i;
    printf("Enter the number of elements:\n");
    scanf("%d", &n);
    printf("Enter the numbers:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &num);
        if (num % 2 == 0) printf("%d is Even.\n", num);
        else printf("%d is Odd.\n", num);
    }
    return 0;
}
